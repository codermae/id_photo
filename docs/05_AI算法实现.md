## 5.1 人脸检测算法

### 5.1.1 face-detection-tflite (BlazeFace)

**技术方案**: face-detection-tflite (fdlite)  
**底层模型**: BlazeFace (Google)  
**运行环境**: CPU (TensorFlow Lite)  
**检测速度**: < 100ms/帧

**核心代码**:
```python
from fdlite import FaceDetection

class AIProcessor:
    def __init__(self):
        # 尝试初始化 face-detection-tflite
        try:
            from fdlite import FaceDetection
            self.face_detector = FaceDetection()
            self.use_tflite = True
            print("[OK] 人脸检测器初始化成功 (fdlite)")
        except Exception as e:
            # 备选方案：OpenCV级联分类器
            self._init_opencv_cascade()
            self.use_tflite = False
    
    def detect_face(self, image):
        """检测人脸"""
        if self.use_tflite and self.face_detector:
            return self._detect_face_tflite(image)
        else:
            return self._detect_face_cascade(image)
    
    def _detect_face_tflite(self, image):
        """使用 fdlite 检测人脸"""
        h, w = image.shape[:2]
        detections = self.face_detector(image)
        
        if detections and len(detections) > 0:
            # 按置信度排序
            detections = sorted(detections, key=lambda d: d.score, reverse=True)
            
            for detection in detections:
                if detection.score >= 0.3:  # 置信度阈值
                    bbox = detection.bbox
                    
                    # 转换坐标（fdlite返回归一化坐标）
                    if bbox.normalized:
                        x = int(bbox.xmin * w)
                        y = int(bbox.ymin * h)
                        x2 = int(bbox.xmax * w)
                        y2 = int(bbox.ymax * h)
                    else:
                        x, y = int(bbox.xmin), int(bbox.ymin)
                        x2, y2 = int(bbox.xmax), int(bbox.ymax)
                    
                    width = x2 - x
                    height = y2 - y
                    
                    # 验证检测框合理性
                    if width > 20 and height > 20:
                        return (x, y, width, height)
        return None
```

**检测流程**:
1. 图像输入fdlite模型
2. BlazeFace进行人脸检测
3. 获取检测结果（边界框+置信度+关键点）
4. 坐标转换（归一化→像素）

**备选方案 - OpenCV级联分类器**:
```python
def _init_opencv_cascade(self):
    """初始化OpenCV级联分类器作为备选"""
    model_dir = cv2.data.haarcascades
    cascade_paths = [
        os.path.join(model_dir, 'lbpcascade_frontalface.xml'),
        os.path.join(model_dir, 'haarcascade_frontalface_alt2.xml'),
        os.path.join(model_dir, 'haarcascade_frontalface_default.xml'),
    ]
    
    for cascade_path in cascade_paths:
        if os.path.exists(cascade_path):
            self.face_cascade = cv2.CascadeClassifier(cascade_path)
            if not self.face_cascade.empty():
                print(f"[OK] 使用级联分类器: {os.path.basename(cascade_path)}")
                break

def _detect_face_cascade(self, image):
    """使用级联分类器检测人脸"""
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    
    faces = self.face_cascade.detectMultiScale(
        gray,
        scaleFactor=1.05,
        minNeighbors=4,
        minSize=(30, 30),
        maxSize=(500, 500)
    )
    
    if len(faces) > 0:
        return max(faces, key=lambda f: f[2] * f[3])
    return None
```

**优化策略**:
- 优先使用fdlite（BlazeFace模型），性能更好
- 置信度阈值设为0.3，提高检测率
- 级联分类器作为备选，保证可用性
- 检测稳定性机制：历史记录平滑

```python
import dlib

class AIProcessor:
    def _init_dlib_landmark_detector(self):
        """初始化dlib人脸关键点检测器"""
        try:
            from face_recognition_models import face_recognition_model_location
            models_dir = os.path.dirname(face_recognition_model_location())
            model_path = os.path.join(models_dir, 'shape_predictor_68_face_landmarks.dat')
            
            if os.path.exists(model_path):
                self.landmark_detector = dlib.shape_predictor(model_path)
                print("[OK] dlib 人脸关键点检测器初始化成功 (68个关键点)")
        except Exception as e:
            print(f"[WARNING] dlib关键点检测器初始化失败: {e}")
            self.landmark_detector = None
    
    def _get_landmarks_from_dlib(self, image, face_box):
        """使用dlib获取人脸关键点"""
        if self.landmark_detector is None:
            return None
        
        x, y, w, h = face_box
        dlib_rect = dlib.rectangle(x, y, x + w, y + h)
        
        landmarks = self.landmark_detector(image, dlib_rect)
        
        landmark_points = []
        for point in landmarks.parts():
            landmark_points.append((point.x, point.y))
        
        return landmark_points
```

```python
def _skin_smoothing(self, image, face_mask, landmarks):
    """智能磨皮，保护重要特征"""
    smooth_strength = self.strength_settings['smooth_strength']
    
    # 创建特征保护mask
    protect_mask = self._create_feature_protection_mask(image.shape[:2], landmarks)
    
    # 双边滤波参数
    d = max(9, int(15 + 20 * smooth_strength))  # 核大小 9-35
    sigma_color = max(30, int(40 + 60 * smooth_strength))
    sigma_space = max(30, int(40 + 60 * smooth_strength))
    iterations = max(1, int(1 + smooth_strength * 2))  # 迭代次数 1-3
    
    # 多次双边滤波
    smoothed = image.copy()
    for i in range(iterations):
        smoothed = cv2.bilateralFilter(smoothed, d, sigma_color, sigma_space)
    
    # 按强度和mask混合
    blend_strength = min(0.8, smooth_strength * 1.0)
    mask_norm = face_mask.astype(np.float32) / 255.0
    
    result = image * (1 - mask_norm * blend_strength) + smoothed * (mask_norm * blend_strength)
    
    return result.astype(np.uint8)
```

## 5.3 背景替换算法

### 5.3.1 rembg (U2-Net) AI语义分割

**技术方案**: rembg + U2-Net模型  
**模型**: u2net (用于人像分割)  
**运行环境**: CPU/GPU (ONNX Runtime)

**核心代码**:
```python
import rembg
from PIL import Image

class PreciseBackgroundReplacer:
    def __init__(self):
        self.rembg = None
        self.rembg_session = None
        self._ensure_rembg_initialized()
    
    def _ensure_rembg_initialized(self):
        """初始化rembg（延迟加载）"""
        try:
            # 支持GPU加速
            providers = ['CPUExecutionProvider']
            try:
                import torch
                if torch.cuda.is_available():
                    providers = ['CUDAExecutionProvider', 'CPUExecutionProvider']
            except:
                pass
            
            self.rembg_session = rembg.new_session('u2net', providers=providers)
            self.rembg_available = True
            print("[INFO] rembg (U2-Net) 已启用")
        except Exception as e:
            print(f"[WARNING] rembg初始化失败: {e}")
    
    def replace_background(self, image, bg_color, method='auto'):
        """背景替换"""
        # 使用rembg进行AI抠图
        mask = self._rembg_segment(image)
        
        # 应用背景
        result = self._apply_background(image, mask, bg_color)
        
        return result
    
    def _rembg_segment(self, image, use_alpha_matting=True):
        """使用rembg进行AI抠图"""
        pil_image = Image.fromarray(cv2.cvtColor(image, cv2.COLOR_BGR2RGB))
        
        # AI抠图 - 启用Alpha Matte提升边缘质量
        if use_alpha_matting:
            output = self.rembg.remove(
                pil_image, 
                session=self.rembg_session,
                alpha_matting=True,
                alpha_matting_foreground_threshold=235,
                alpha_matting_background_threshold=5,
                alpha_matting_erode_size=15
            )
        else:
            output = self.rembg.remove(pil_image, session=self.rembg_session)
        
        # 提取alpha通道作为mask
        output_array = np.array(output)
        if output_array.shape[2] == 4:  # RGBA
            mask = output_array[:, :, 3]  # Alpha通道
        
        return mask
    
    def _apply_background(self, image, mask, bg_color):
        """应用背景颜色"""
        h, w = image.shape[:2]
        background = np.full((h, w, 3), bg_color, dtype=np.uint8)
        
        # Alpha混合
        mask_norm = mask.astype(np.float32) / 255.0
        mask_3ch = np.stack([mask_norm, mask_norm, mask_norm], axis=2)
        
        result = (image * mask_3ch + background * (1 - mask_3ch)).astype(np.uint8)
        
        return result
```

**rembg优势**:
- 基于U2-Net深度学习模型
- 头发丝级精细抠图
- 支持Alpha Matte边缘优化
- 支持GPU加速

**Alpha Matte参数**:
- `alpha_matting_foreground_threshold`: 235（前景阈值）
- `alpha_matting_background_threshold`: 5（背景阈值）
- `alpha_matting_erode_size`: 15（腐蚀大小）

## 5.4 智能裁剪算法

### 5.4.1 人脸定位裁剪

**算法流程**:
```
检测人脸位置
    ↓
估算完整头部高度（含头发）
    ↓
计算裁剪区域（头顶留白30%）
    ↓
人脸居中定位
    ↓
裁剪并缩放到目标尺寸
```

**核心代码**:
```python
class SmartCropper:
    def __init__(self):
        # 证件照规格定义 (像素尺寸, DPI=600)
        self.crop_specs = {
            '一寸': {'width': 590, 'height': 826, 'mm_width': 25, 'mm_height': 35},

            '美国护照': {'width': 1200, 'height': 1200, 'mm_width': 51, 'mm_height': 51},
            '欧盟护照': {'width': 826, 'height': 1063, 'mm_width': 35, 'mm_height': 45},
            # ... 更多规格
        }
    
    def smart_crop(self, image, spec='一寸', face_detector=None):
        """智能裁剪"""
        spec_info = self.crop_specs[spec]
        h, w = image.shape[:2]
        
        # 检测人脸
        face_info = None
        if face_detector:
            face_info = self._detect_face_info(image, face_detector)
        
        if face_info:
            crop_area = self._calculate_face_based_crop(image.shape, face_info, spec_info)
        else:
            crop_area = self._calculate_center_crop(image.shape, spec_info)
        
        # 执行裁剪
        cropped = image[crop_area[1]:crop_area[1]+crop_area[3], 
                       crop_area[0]:crop_area[0]+crop_area[2]]
        
        # 调整到目标尺寸
        final = cv2.resize(cropped, (spec_info['width'], spec_info['height']),
                          interpolation=cv2.INTER_AREA)
        
        return final
```

**基于人脸的裁剪计算**:
```python
def _calculate_face_based_crop(self, image_shape, face_info, spec_info):
    """基于人脸位置计算裁剪区域"""
    h, w = image_shape[:2]
    target_ratio = spec_info['width'] / spec_info['height']
    
    face_width = face_info['width']
    face_height = face_info['height']
    face_top = face_info['top']
    face_bottom = face_info['bottom']
    
    # 1. 估算完整头部高度（包括头发）
    # 人脸检测通常只检测到眉毛到下巴，头发在上面
    estimated_hair_height = int(face_height * 0.4)
    full_head_top = max(0, face_top - estimated_hair_height)
    full_head_height = face_bottom - full_head_top
    
    # 2. 根据证件照标准计算照片高度
    # 头顶留白：头部高度的35%
    # 下巴留白：头部高度的25%
    top_margin = int(full_head_height * 0.35)
    bottom_margin = int(full_head_height * 0.25)
    
    estimated_photo_height = full_head_height + top_margin + bottom_margin
    estimated_photo_width = int(estimated_photo_height * target_ratio)
    
    # 3. 计算裁剪位置
    crop_y = max(0, full_head_top - top_margin)
    crop_x = max(0, min(w - estimated_photo_width, 
                        face_info['center'][0] - estimated_photo_width // 2))
    
    return (crop_x, crop_y, estimated_photo_width, estimated_photo_height)
```

**支持的规格**（15种）:
- 中国标准：一寸、二寸、小二寸、大一寸、驾驶证、社保卡
- 国际标准：美国护照、欧盟护照、英国签证、日本护照、印度签证、泰国签证
- 大尺寸：五寸、六寸

## 5.5 人脸识别与身份核验

### 5.5.1 face-recognition (dlib)

**技术方案**: face-recognition库（基于dlib）  
**特征维度**: 128维人脸特征向量  
**相似度计算**: 欧氏距离

**核心代码**:
```python
import face_recognition

class DuplicateChecker:
    def __init__(self):
        pass
    
    def encode_face(self, image):
        """提取人脸特征向量"""
        # face_recognition使用RGB图像
        rgb_image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
        
        # 检测人脸位置
        face_locations = face_recognition.face_locations(rgb_image)
        
        if len(face_locations) == 0:
            return None
        
        # 提取特征向量（128维）
        encodings = face_recognition.face_encodings(rgb_image, face_locations)
        
        return encodings[0]  # 返回第一个人脸的特征
    
    def check_duplicate(self, new_encoding, existing_encoding, threshold=0.6):
        """检查是否重复"""
        # 计算欧氏距离
        distance = np.linalg.norm(new_encoding - existing_encoding)
        
        # 转换为相似度
        similarity = 1.0 - min(distance / 2.0, 1.0)
        
        # 判断是否为同一人
        is_duplicate = similarity >= threshold
        
        return {
            'is_duplicate': is_duplicate,
            'similarity': similarity,
            'distance': distance
        }
```

**相似度计算公式**:
$$\text{Similarity} = 1 - \min\left(\frac{d}{2}, 1\right)$$

其中 $d$ 为欧氏距离。

**阈值设置**:
- 0.6: 默认阈值（推荐）
- 0.5: 更严格
- 0.7: 更宽松

## 5.6 质量评分算法

### 5.6.1 多维度评分

**评分维度**:
1. **人脸位置** - 是否居中
2. **人脸大小** - 是否合适
3. **光照条件** - 是否均匀
4. **眼睛状态** - 是否正视
5. **嘴巴状态** - 是否闭合
6. **是否戴眼镜** - 检测眼镜

**核心代码**:
```python
def check_face_quality(self, image):
    """检查人脸质量"""
    quality_info = {
        'face_detected': False,
        'is_frontal': False,
        'eyes_open': False,
        'eyes_looking_forward': False,
        'mouth_closed': False,
        'no_glasses': False,
        'good_lighting': False,
        'face_position': 'center',
        'face_size': 'good',
        'overall_score': 0
    }
    
    face = self.detect_face(image)
    if face is None:
        return quality_info
    
    quality_info['face_detected'] = True
    x, y, fw, fh = face
    h, w = image.shape[:2]
    
    # 1. 人脸位置检查
    face_center_x = x + fw / 2
    face_center_y = y + fh / 2
    image_center_x = w / 2
    image_center_y = h / 2
    
    h_tolerance = w * 0.1
    v_tolerance = h * 0.08
    
    if abs(face_center_x - image_center_x) <= h_tolerance and \
       abs(face_center_y - image_center_y) <= v_tolerance:
        quality_info['is_frontal'] = True
        quality_info['face_position'] = 'center'
    
    # 2. 人脸大小检查
    face_area = fw * fh
    image_area = w * h
    face_ratio = face_area / image_area
    
    if face_ratio < 0.05:
        quality_info['face_size'] = 'too_small'
    elif face_ratio > 0.3:
        quality_info['face_size'] = 'too_large'
    else:
        quality_info['face_size'] = 'good'
    
    # 3. 光照检查
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    face_roi = gray[y:y+fh, x:x+fw]
    brightness = np.mean(face_roi)
    brightness_std = np.std(face_roi)
    
    if 60 < brightness < 180 and brightness_std > 20:
        quality_info['good_lighting'] = True
    
    # 4. 眼睛检查
    quality_info['eyes_looking_forward'] = self._check_eyes_looking_forward(image, face)
    
    # 5. 嘴巴检查
    quality_info['mouth_closed'] = self._check_mouth_closed(image, face)
    
    # 6. 眼镜检查
    quality_info['no_glasses'] = self._check_no_glasses(image, face)
    
    # 综合评分
    score = 0
    if quality_info['face_detected']: score += 20
    if quality_info['is_frontal']: score += 20
    if quality_info['face_size'] == 'good': score += 15
    if quality_info['good_lighting']: score += 15
    if quality_info['eyes_looking_forward']: score += 15
    if quality_info['mouth_closed']: score += 10
    if quality_info['no_glasses']: score += 5
    
    quality_info['overall_score'] = score
    
    return quality_info
```

**评分标准**:
- 90-100分: 优秀
- 80-89分: 良好
- 70-79分: 合格
- < 70分: 需重拍

```python
self.history_size = 5
self.stable_threshold = 3

def detect_face(self, image):
    """检测人脸 - 带稳定性机制"""
    raw_result = self._detect_face_tflite(image)
    
    # 更新检测历史
    self.detection_history.append(raw_result is not None)
    if len(self.detection_history) > self.history_size:
        self.detection_history.pop(0)
    
    # 计算稳定性
    if len(self.detection_history) >= 3:
        positive_count = sum(self.detection_history)
        if positive_count >= self.stable_threshold:
            return raw_result
        elif positive_count <= 1:
            return None
    
    return raw_result
```
