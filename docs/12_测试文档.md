### 图像处理模块测试

#### test_image_processor.py

```python
import pytest
import cv2
import numpy as np
from controllers.image_processor import ImageProcessor

class TestImageProcessor:
    """图像处理控制器测试"""
    
    @pytest.fixture
    def processor(self):
        """创建处理器实例"""
        return ImageProcessor()
    
    @pytest.fixture
    def sample_image(self):
        """创建示例图像"""
        return np.random.randint(0, 256, (1080, 1440, 3), dtype=np.uint8)
    
    def test_load_image(self, processor, tmp_path):
        """测试加载图像"""
        # 创建临时图像文件
        img_path = tmp_path / "test.jpg"
        cv2.imwrite(str(img_path), np.zeros((100, 100, 3), dtype=np.uint8))
        
        # 加载图像
        image = processor.load_image(str(img_path))
        
        # 验证
        assert image is not None
        assert image.shape == (100, 100, 3)
    
    def test_crop_to_spec(self, processor, sample_image):
        """测试按规格裁剪"""
        processor.current_image = sample_image
        
        # 测试各种规格
        specs = ['一寸', '二寸', '小二寸']
        for spec in specs:
            success, info = processor.crop_to_spec(spec)
            assert success or 'error' in info
    
    def test_change_background(self, processor, sample_image):
        """测试背景替换"""
        processor.current_image = sample_image
        
        # 测试各种背景色
        colors = ['白色', '蓝色', '红色']
        for color in colors:
            success = processor.change_background(color)
            assert isinstance(success, bool)
    
    def test_beautify(self, processor, sample_image):
        """测试美颜处理"""
        processor.current_image = sample_image
        
        # 测试不同强度
        for strength in [0.5, 1.0, 1.5]:
            success = processor.beautify(strength)
            assert isinstance(success, bool)
    
    def test_undo_redo(self, processor, sample_image):
        """测试撤销/重做"""
        processor.current_image = sample_image
        original = processor.current_image.copy()
        
        # 执行操作
        processor.adjust_brightness(10)
        modified = processor.current_image.copy()
        
        # 撤销
        processor.undo()
        assert np.array_equal(processor.current_image, original)
        
        # 重做
        processor.redo()
        assert np.array_equal(processor.current_image, modified)
```

### 完整工作流测试

#### test_integration.py

```python
import pytest
from datetime import date
from controllers.image_processor import ImageProcessor
from controllers.ai_processor import AIProcessor
from utils.database_helper import DatabaseHelper

class TestIntegration:
    """集成测试"""
    
    @pytest.fixture
    def setup(self):
        """设置测试环境"""
        ai_processor = AIProcessor()
        image_processor = ImageProcessor(ai_processor=ai_processor)
        db_helper = DatabaseHelper()
        
        return {
            'ai_processor': ai_processor,
            'image_processor': image_processor,
            'db_helper': db_helper
        }
    
    def test_complete_workflow(self, setup, sample_image_path):
        """测试完整工作流"""
        ai = setup['ai_processor']
        processor = setup['image_processor']
        db = setup['db_helper']
        
        # 1. 添加用户
        user = db.add_user(
            name='测试用户',
            id_number='110101199001011238'
        )
        
        # 2. 加载图像
        processor.load_image(sample_image_path)
        
        # 3. 检查质量
        quality = ai.check_face_quality(processor.current_image)
        assert quality['score'] >= 0
        
        # 4. 处理图像
        processor.crop_to_spec('一寸')
        processor.change_background('白色')
        processor.beautify()
        
        # 5. 保存结果
        success, path = processor.save_processed_image(
            user.id_number, '一寸', '白色'
        )
        assert success
        
        # 6. 记录照片
        photo = db.add_photo(
            user_id=user.id,
            photo_type='processed',
            file_path=path,
            quality_score=quality['score']
        )
        assert photo.id is not None
        
        # 7. 记录采集
        record = db.add_collection_record(
            user_id=user.id,
            collection_date=date.today(),
            status='completed'
        )
        assert record.id is not None
```

## 常见问题

### Q1: 测试失败怎么办？

**A**: 检查以下几点：
1. 依赖是否正确安装
2. 测试数据是否完整
3. 环境变量是否设置
4. 数据库是否初始化

### Q2: 如何调试测试？

**A**: 使用以下方法：
```bash
# 添加-s参数显示print输出
pytest -s

# 使用pdb调试
pytest --pdb

# 显示局部变量
pytest -l
```

### Q3: 如何跳过某些测试？

**A**: 使用skip装饰器：
```python
@pytest.mark.skip(reason="暂不实现")
def test_feature():
    pass
```
