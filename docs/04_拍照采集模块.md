# 拍照采集模块

## 模块概述

拍照采集模块是系统的核心功能之一，负责通过摄像头实时采集用户照片。该模块集成了AI人脸检测、图像质量评估、实时预览等功能，确保采集的照片符合证件照标准。

### 1. 摄像头控制
- 自动检测可用摄像头
- 支持多摄像头切换
- 实时视频流预览
- 分辨率自适应调整

### 2. AI质量检测
- 实时人脸检测

- 语音提示（可选）

### 4. 自动保存
- 自动生成文件名
- 自动保存到指定目录
- 自动更新采集记录
- 自动关联用户信息

### 摄像头控制类

```python
class Camera:
    """摄像头控制类"""
    
    def __init__(self, camera_id: int = 0):
        """
        初始化摄像头
        
        Args:
            camera_id: 摄像头ID（0为默认摄像头）
        """
        self.camera_id = camera_id
        self.cap = None
        self.is_opened = False
        self.frame_width = 1280
        self.frame_height = 720
    
    def open(self) -> bool:
        """
        打开摄像头
        
        Returns:
            bool: 是否成功打开
        """
        try:
            self.cap = cv2.VideoCapture(self.camera_id)
            
            if not self.cap.isOpened():
                return False
            
            # 设置分辨率
            self.cap.set(cv2.CAP_PROP_FRAME_WIDTH, self.frame_width)
            self.cap.set(cv2.CAP_PROP_FRAME_HEIGHT, self.frame_height)
            
            # 设置自动对焦
            self.cap.set(cv2.CAP_PROP_AUTOFOCUS, 1)
            
            self.is_opened = True
            return True
            
        except Exception as e:
            logger.error(f"打开摄像头失败: {str(e)}")
            return False
    
    def read_frame(self) -> tuple[bool, np.ndarray]:
        """
        读取一帧图像
        
        Returns:
            tuple: (是否成功, 图像数据)
        """
        if not self.is_opened:
            return False, None
        
        ret, frame = self.cap.read()
        return ret, frame
    
    def close(self):
        """关闭摄像头"""
        if self.cap is not None:
            self.cap.release()
            self.is_opened = False
    
    @staticmethod
    def get_available_cameras() -> list:
        """
        获取可用摄像头列表
        
        Returns:
            list: 可用摄像头ID列表
        """
        available = []
        for i in range(10):  # 检查前10个设备
            cap = cv2.VideoCapture(i)
            if cap.isOpened():
                available.append(i)
                cap.release()
        return available
```

### 摄像头线程

```python
class CameraThread(QThread):
    """摄像头采集线程"""
    
    frame_ready = pyqtSignal(np.ndarray)  # 帧就绪信号
    quality_result = pyqtSignal(dict)     # 质量检测结果信号
    
    def __init__(self, camera: Camera, ai_processor: AIProcessor):
        super().__init__()
        self.camera = camera
        self.ai_processor = ai_processor
        self.running = False
    
    def run(self):
        """线程运行"""
        self.running = True
        
        while self.running:
            # 读取一帧
            ret, frame = self.camera.read_frame()
            
            if not ret:
                break
            
            # 发送帧数据
            self.frame_ready.emit(frame.copy())
            
            # AI质量检测
            quality = self.ai_processor.analyze_quality(frame)
            self.quality_result.emit(quality)
            
            # 控制帧率
            self.msleep(33)  # 约30fps
    
    def stop(self):
        """停止线程"""
        self.running = False
        self.wait()
```

### 人脸检测

```python
def detect_face(self, image: np.ndarray) -> dict:
    """
    检测人脸
    
    Args:
        image: 输入图像
        
    Returns:
        dict: {
            'detected': bool,      # 是否检测到人脸
            'count': int,          # 人脸数量
            'bbox': tuple,         # 人脸边界框 (x, y, w, h)
            'confidence': float,   # 置信度
            'landmarks': list      # 关键点坐标
        }
    """
    # 转换为RGB
    rgb_image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    
    # MediaPipe人脸检测
    results = self.face_detection.process(rgb_image)
    
    if not results.detections:
        return {
            'detected': False,
            'count': 0,
            'bbox': None,
            'confidence': 0.0,
            'landmarks': []
        }
    
    # 获取第一个检测到的人脸
    detection = results.detections[0]
    
    # 提取边界框
    bbox = detection.location_data.relative_bounding_box
    h, w, _ = image.shape
    x = int(bbox.xmin * w)
    y = int(bbox.ymin * h)
    width = int(bbox.width * w)
    height = int(bbox.height * h)
    
    # 提取关键点
    landmarks = []
    for landmark in detection.location_data.relative_keypoints:
        landmarks.append((int(landmark.x * w), int(landmark.y * h)))
    
    return {
        'detected': True,
        'count': len(results.detections),
        'bbox': (x, y, width, height),
        'confidence': detection.score[0],
        'landmarks': landmarks
    }
```

### 光线检测

```python
def check_brightness(self, image: np.ndarray) -> dict:
    """
    检测光线亮度
    
    Args:
        image: 输入图像
        
    Returns:
        dict: {
            'brightness': float,   # 亮度值 (0-255)
            'status': str,         # 状态: 'good', 'dark', 'bright'
            'message': str         # 提示信息
        }
    """
    # 转换为灰度图
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    
    # 计算平均亮度
    brightness = np.mean(gray)
    
    # 判断亮度状态
    if brightness < 80:
        status = 'dark'
        message = '光线太暗，请增加光线'
    elif brightness > 200:
        status = 'bright'
        message = '光线太亮，请减弱光线'
    else:
        status = 'good'
        message = '光线适中'
    
    return {
        'brightness': float(brightness),
        'status': status,
        'message': message
    }
```

### 清晰度检测

```python
def check_sharpness(self, image: np.ndarray) -> dict:
    """
    检测图像清晰度
    
    Args:
        image: 输入图像
        
    Returns:
        dict: {
            'sharpness': float,    # 清晰度值
            'status': str,         # 状态: 'sharp', 'blur'
            'message': str         # 提示信息
        }
    """
    # 转换为灰度图
    gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
    
    # 使用Laplacian算子计算清晰度
    laplacian = cv2.Laplacian(gray, cv2.CV_64F)
    sharpness = laplacian.var()
    
    # 判断清晰度状态
    if sharpness < 100:
        status = 'blur'
        message = '图像模糊，请保持稳定'
    else:
        status = 'sharp'
        message = '图像清晰'
    
    return {
        'sharpness': float(sharpness),
        'status': status,
        'message': message
    }
```

### 综合质量评估

```python
def analyze_quality(self, image: np.ndarray) -> dict:
    """
    综合质量评估
    
    Args:
        image: 输入图像
        
    Returns:
        dict: {
            'overall': str,        # 总体评价: 'excellent', 'good', 'poor'
            'score': float,        # 质量分数 (0-100)
            'face': dict,          # 人脸检测结果
            'brightness': dict,    # 亮度检测结果
            'sharpness': dict,     # 清晰度检测结果
            'suggestions': list    # 改进建议
        }
    """
    # 人脸检测
    face_result = self.detect_face(image)
    
    # 光线检测
    brightness_result = self.check_brightness(image)
    
    # 清晰度检测
    sharpness_result = self.check_sharpness(image)
    
    # 计算质量分数
    score = 0
    suggestions = []
    
    # 人脸检测 (40分)
    if face_result['detected']:
        if face_result['count'] == 1:
            score += 40
        else:
            score += 20
            suggestions.append('画面中应只有一个人')
    else:
        suggestions.append('未检测到人脸')
    
    # 光线亮度 (30分)
    if brightness_result['status'] == 'good':
        score += 30
    else:
        score += 10
        suggestions.append(brightness_result['message'])
    
    # 图像清晰度 (30分)
    if sharpness_result['status'] == 'sharp':
        score += 30
    else:
        score += 10
        suggestions.append(sharpness_result['message'])
    
    # 总体评价
    if score >= 90:
        overall = 'excellent'
    elif score >= 70:
        overall = 'good'
    else:
        overall = 'poor'
    
    return {
        'overall': overall,
        'score': score,
        'face': face_result,
        'brightness': brightness_result,
        'sharpness': sharpness_result,
        'suggestions': suggestions
    }
```

## 拍照流程

### 完整流程

```python
def take_photo(self):
    """拍照"""
    # 1. 检查摄像头状态
    if not self.camera.is_opened:
        QMessageBox.warning(self, '警告', '请先打开摄像头')
        return
    
    # 2. 获取当前帧
    ret, frame = self.camera.read_frame()
    if not ret:
        QMessageBox.warning(self, '警告', '获取图像失败')
        return
    
    # 3. 质量检查
    quality = self.ai_processor.analyze_quality(frame)
    if quality['score'] < 70:
        reply = QMessageBox.question(
            self, '质量提示',
            f'当前质量评分: {quality["score"]}分\n'
            f'建议: {", ".join(quality["suggestions"])}\n\n'
            f'是否继续拍照？',
            QMessageBox.Yes | QMessageBox.No
        )
        if reply == QMessageBox.No:
            return
    
    # 4. 选择用户
    user = self.select_user()
    if not user:
        return
    
    # 5. 保存照片
    filepath = FileHelper.save_raw_photo(
        user.id_number,
        frame
    )
    
    # 6. 更新数据库
    db = Database()
    
    # 添加照片记录
    db.add_photo(
        user_id=user.id,
        filepath=filepath,
        photo_type='raw'
    )
    
    # 更新采集记录
    record = db.get_collection_record_by_user(user.id)
    if record:
        db.update_collection_status(record.id, 'completed')
    
    db.close()
    
    # 7. 显示成功提示
    QMessageBox.information(
        self, '成功',
        f'照片已保存\n\n'
        f'用户: {user.name}\n'
        f'路径: {filepath}'
    )
    
    # 8. 通知主窗口更新
    self.notify_data_changed()
```

```python
def select_user(self) -> User:
    """选择用户"""
    db = Database()
    
    # 获取待采集用户列表
    pending_users = db.get_users_by_status('pending')
    
    if not pending_users:
        QMessageBox.information(self, '提示', '没有待采集的用户')
        db.close()
        return None
    
    # 显示选择对话框
    dialog = UserSelectDialog(pending_users)
    if dialog.exec_() == QDialog.Accepted:
        user = dialog.selected_user
        db.close()
        return user
    
    db.close()
    return None
```

## 性能优化

### 帧率控制

```python
class FrameRateController:
    """帧率控制器"""
    
    def __init__(self, target_fps: int = 30):
        self.target_fps = target_fps
        self.frame_time = 1.0 / target_fps
        self.last_time = time.time()
    
    def wait(self):
        """等待到下一帧"""
        current_time = time.time()
        elapsed = current_time - self.last_time
        
        if elapsed < self.frame_time:
            time.sleep(self.frame_time - elapsed)
        
        self.last_time = time.time()
```

### 图像缓存

```python
class FrameBuffer:
    """帧缓冲区"""
    
    def __init__(self, size: int = 5):
        self.buffer = []
        self.size = size
    
    def add(self, frame: np.ndarray):
        """添加帧"""
        self.buffer.append(frame.copy())
        if len(self.buffer) > self.size:
            self.buffer.pop(0)
    
    def get_best_frame(self) -> np.ndarray:
        """获取质量最好的帧"""
        if not self.buffer:
            return None
        
        # 评估每一帧的质量
        scores = []
        for frame in self.buffer:
            quality = self.ai_processor.analyze_quality(frame)
            scores.append(quality['score'])
        
        # 返回质量最高的帧
        best_index = scores.index(max(scores))
        return self.buffer[best_index]
```

## 常见问题

### Q1: 摄像头打不开？

**原因**：
- 摄像头被其他程序占用
- 驱动问题
- 权限问题

**解决**：
1. 关闭其他使用摄像头的程序
2. 更新摄像头驱动
3. 检查应用权限设置

### Q2: 画面卡顿？

**原因**：
- CPU占用过高
- 分辨率过高
- 帧率设置不当

**解决**：
1. 降低分辨率
2. 降低帧率
3. 关闭其他程序

### Q3: 人脸检测不准确？

**原因**：
- 光线不足
- 角度不正
- 遮挡

**解决**：
1. 改善光线条件
2. 调整拍摄角度
3. 移除遮挡物
