# 数据导入模块

## 模块概述

数据导入模块提供手动录入和批量导入两种方式，用于将用户信息录入系统。该模块支持单个用户手动添加和Excel批量导入，并提供完善的数据验证功能。

---

## 功能特性

### 1. 手动录入

- 表单式信息录入
- 实时数据验证
- 自动识别性别和年龄
- 支持快速录入

### 2. 批量导入

- Excel文件导入
- 批量数据验证
- 导入进度显示
- 错误数据提示

### 3. 数据验证

- 身份证号格式验证
- 校验码验证
- 重复数据检测
- 必填项检查

### 4. 自动功能

- 自动创建采集记录
- 自动更新系统状态
- 自动刷新用户列表

### 数据模型

```python
class UserInput:
    """用户输入数据模型"""
    
    def __init__(self):
        self.name: str = ""
        self.id_number: str = ""
        self.gender: str = ""
        self.phone: str = ""
    
    def validate(self) -> tuple[bool, str]:
        """验证数据"""
        # 1. 必填项检查
        if not self.name:
            return False, "姓名不能为空"
        
        if not self.id_number:
            return False, "身份证号不能为空"
        
        # 2. 身份证号验证
        result = IDCardValidator.validate(self.id_number)
        if not result['valid']:
            return False, result['message']
        
        # 3. 电话号码验证（可选）
        if self.phone and not self.validate_phone():
            return False, "电话号码格式不正确"
        
        return True, "验证通过"
    
    def validate_phone(self) -> bool:
        """验证电话号码"""
        import re
        pattern = r'^1[3-9]\d{9}$'
        return bool(re.match(pattern, self.phone))
```

### 添加用户

```python
def add_user(self):
    """添加用户"""
    # 1. 获取输入数据
    user_input = UserInput()
    user_input.name = self.name_input.text().strip()
    user_input.id_number = self.id_number_input.text().strip()
    user_input.gender = self.gender_combo.currentText()
    user_input.phone = self.phone_input.text().strip()
    
    # 2. 验证数据
    valid, message = user_input.validate()
    if not valid:
        QMessageBox.warning(self, '验证失败', message)
        return
    
    # 3. 检查重复
    db = Database()
    existing = db.get_user_by_id_number(user_input.id_number)
    if existing:
        reply = QMessageBox.question(
            self, '确认',
            f'用户 {existing.name} 已存在，是否更新？',
            QMessageBox.Yes | QMessageBox.No
        )
        if reply == QMessageBox.No:
            db.close()
            return
        
        # 更新用户
        db.update_user(
            existing.id,
            name=user_input.name,
            gender=user_input.gender,
            phone=user_input.phone
        )
        QMessageBox.information(self, '成功', '用户信息已更新')
    else:
        # 添加新用户
        user = db.add_user(
            name=user_input.name,
            id_number=user_input.id_number,
            gender=user_input.gender,
            phone=user_input.phone
        )
        
        # 创建采集记录
        db.add_collection_record(
            user_id=user.id,
            status='pending'
        )
        
        QMessageBox.information(self, '成功', '用户添加成功')
    
    db.close()
    
    # 4. 清空表单
    self.clear_form()
    
    # 5. 通知主窗口更新
    self.notify_data_changed()
```

### 读取Excel

```python
def read_excel_file(self, filepath: str) -> list:
    """读取Excel文件"""
    import pandas as pd
    
    try:
        # 读取Excel
        df = pd.read_excel(filepath)
        
        # 检查必需列
        required_columns = ['姓名', '身份证号']
        for col in required_columns:
            if col not in df.columns:
                raise ValueError(f'缺少必需列: {col}')
        
        # 转换为字典列表
        users = []
        for index, row in df.iterrows():
            user = {
                'name': str(row['姓名']).strip(),
                'id_number': str(row['身份证号']).strip(),
                'gender': str(row.get('性别', '')).strip(),
                'phone': str(row.get('电话', '')).strip()
            }
            users.append(user)
        
        return users
    
    except Exception as e:
        raise Exception(f'读取Excel失败: {str(e)}')
```

### 批量验证

```python
def validate_batch_users(self, users: list) -> tuple[list, list]:
    """
    批量验证用户数据
    
    Returns:
        tuple: (有效用户列表, 错误列表)
    """
    valid_users = []
    errors = []
    
    for i, user in enumerate(users):
        row_num = i + 2  # Excel行号（从2开始，因为第1行是标题）
        
        # 1. 必填项检查
        if not user['name']:
            errors.append(f'第{row_num}行: 姓名不能为空')
            continue
        
        if not user['id_number']:
            errors.append(f'第{row_num}行: 身份证号不能为空')
            continue
        
        # 2. 身份证号验证
        result = IDCardValidator.validate(user['id_number'])
        if not result['valid']:
            errors.append(f'第{row_num}行: {result["message"]}')
            continue
        
        # 3. 自动填充性别（如果未提供）
        if not user['gender']:
            user['gender'] = result['info']['gender']
        
        # 4. 电话号码验证（如果提供）
        if user['phone']:
            if not self.validate_phone(user['phone']):
                errors.append(f'第{row_num}行: 电话号码格式不正确')
                continue
        
        valid_users.append(user)
    
    return valid_users, errors
```

```python
def validate_phone(self, phone: str) -> bool:
    """
    验证手机号码
    
    规则:
    - 11位数字
    - 以1开头
    - 第二位是3-9
    """
    import re
    
    # 去除空格和连字符
    phone = phone.replace(' ', '').replace('-', '')
    
    # 正则表达式验证
    pattern = r'^1[3-9]\d{9}$'
    
    return bool(re.match(pattern, phone))
```

### 姓名验证

```python
def validate_name(self, name: str) -> bool:
    """
    验证姓名
    
    规则:
    - 2-20个字符
    - 只能包含中文、英文字母
    - 可以包含·（用于少数民族姓名）
    """
    import re
    
    # 长度检查
    if len(name) < 2 or len(name) > 20:
        return False
    
    # 字符检查
    pattern = r'^[\u4e00-\u9fa5a-zA-Z·]+$'
    
    return bool(re.match(pattern, name))
```

## 错误处理

### 导入错误处理

```python
def handle_import_errors(self, errors: list):
    """处理导入错误"""
    if not errors:
        return
    
    # 创建错误报告
    error_report = "导入过程中发现以下错误:\n\n"
    for error in errors[:10]:  # 最多显示10个错误
        error_report += f"• {error}\n"
    
    if len(errors) > 10:
        error_report += f"\n... 还有 {len(errors) - 10} 个错误"
    
    # 显示错误对话框
    QMessageBox.warning(self, '导入错误', error_report)
    
    # 保存完整错误日志
    self.save_error_log(errors)
```

### 错误日志

```python
def save_error_log(self, errors: list):
    """保存错误日志"""
    import datetime
    
    timestamp = datetime.datetime.now().strftime('%Y%m%d_%H%M%S')
    log_file = f'data/exports/import_errors_{timestamp}.txt'
    
    with open(log_file, 'w', encoding='utf-8') as f:
        f.write(f"导入错误日志\n")
        f.write(f"时间: {datetime.datetime.now()}\n")
        f.write(f"错误数量: {len(errors)}\n\n")
        
        for error in errors:
            f.write(f"{error}\n")
    
    logger.info(f"错误日志已保存: {log_file}")
```

## 常见问题

### Q1: Excel导入失败？

**原因**：
- Excel格式不正确
- 缺少必需列
- 数据格式错误

**解决**：
1. 下载并使用标准模板
2. 检查列名是否正确
3. 检查数据格式

### Q2: 身份证号总是提示无效？

**原因**：
- 身份证号格式错误
- 校验码不正确
- 包含空格或特殊字符

**解决**：
1. 检查身份证号长度（18位）
2. 使用`generate_valid_id.py`生成测试数据
3. 去除空格和特殊字符

### Q3: 批量导入很慢？

**原因**：
- 数据量大
- 逐条插入效率低

**解决**：
1. 使用批量插入
2. 开启事务
3. 分批导入
